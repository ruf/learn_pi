Lesson 11: Guess?
-----------------
import random

MIN, MAX = 1, 100
print("I have a number between %d and %d, can you guess it?" %(MIN, MAX))
n = random.randint(MIN, MAX)
for i in range(1, 10):
  x = input("Guess %d ? " % i)
  if x == n:
    print(" - Bingo!")
    break
  if x > n:
    print(" - Too big ...")
  else:
    print(" - Too small ...")
